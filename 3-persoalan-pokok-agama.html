```python

# Informasi pegawai

pegawai1 = {

'nama': 'Ahmad Muslim',

'agama': 'Islam',

'gaji_pokok': 4000000,

'jumlah_anak': 2

}

pegawai2 = {

'nama': 'Alex',

'agama': 'Kristen Protestan',

'gaji_pokok': 6000000,

'jumlah_anak': 5

}

# Fungsi untuk menghitung tunjangan jabatan

def hitung_tunjangan_jabatan(gaji_pokok):

return 0.2 * gaji_pokok

# Fungsi untuk menghitung tunjangan keluarga

def hitung_tunjangan_keluarga(gaji_pokok, jumlah_anak):

if jumlah_anak <= 2:

return 0.1 * gaji_pokok

else:

return 0.2 * gaji_pokok

# Fungsi untuk menghitung gaji bersih

def hitung_gaji_bersih(gaji_pokok, tunjangan_jabatan, tunjangan_keluarga):

return gaji_pokok + tunjangan_jabatan + tunjangan_keluarga

# Menampilkan informasi pegawai dan hasil perhitungan

def cetak_info_pegawai(pegawai):

print(f'Nama Pegawai: {pegawai["nama"]}')

print(f'Agama: {pegawai["agama"]}')

print(f'Gaji Pokok: Rp. {pegawai["gaji_pokok"]:,}')

tunjangan_jabatan = hitung_tunjangan_jabatan(pegawai["gaji_pokok"])

print(f'Tunjangan Jabatan: Rp. {tunjangan_jabatan:,}')

print(f'Tunjangan Keluarga: Rp. {tunjangan_keluarga:,}')

print(f'Gaji Bersih (Take Home Pay): Rp. {gaji_bersih:,}')

# Menampilkan informasi pegawai

cetak_info_pegawai(pegawai1)
```
